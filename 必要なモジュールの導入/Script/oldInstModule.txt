function Chek-EnvModule{
    param(
        [string]$ModuleName,
        [string]$ModuleVersion
    )
    # モジュー ルがインストール済みか確認
    $cnt = ((Get-Module -ListAvailable -Name $ModuleName).Name).Count
    $flg = $false   # yaml記述バージョンのモジュール存在フラグ
    if(!($cnt -eq 0)){
        $getVer = (Get-Module -ListAvailable -Name $ModuleName).Version
        foreach($ver in $getVer){
            if($ver.ToString() -eq $ModuleVersion){
                $flg = $true    # 存在したらフラグを立てる
                Log-Output ("INFO: "+(Get-Date -Format "yyyyMMdd HH:mm:ss")+" [EXIST] "+$ModuleName+" Version: "+$ver.ToString())
            }else{
                Log-Output ("INFO: "+(Get-Date -Format "yyyyMMdd HH:mm:ss")+" [OTHER] "+$ModuleName+" Version: "+$ver.ToString())
            }
        }
    }else{
        Log-Output ("WARN: "+(Get-Date -Format "yyyyMMdd HH:mm:ss")+" [NOTHING] "+$ModuleName)
    }
    # yaml記述バージョンのモジュールが存在しない場合はインストール
    if($flg -eq $false){
        Log-Output ("WARN: "+(Get-Date -Format "yyyyMMdd HH:mm:ss")+" [INSTALL] "+$ModuleName+" Version: "+$ModuleVersion.ToString())
        Install-Module -Name $ModuleName -RequiredVersion $ModuleVersion -Force # | Tee-Object -FilePath $glbLog -Append | Out-Default
    }
}
